For ciklus I. 
    
    
      # A for ciklus egy folyamatvezérlő utasítás
      
      # Adott egy gyumi tömb
      
      gyumi = ['alma', 'körte', 'szilva', 'barack']
      
      #A for ciklus segítségével kiírjuk a gyumi tömb elemeit sorba
      # Az i a ciklus indexét is jelölheti azaz folyamatosan növeli magát 1-gyel!(Erről később bővebben lesz infó)
      # A range() jelöli, hogy meddig menjen a ciklus
      # A len(gyumi) megadja az adott tömb hosszát
      for i in range(len(gyumi)):
        print(gyumi[i])
