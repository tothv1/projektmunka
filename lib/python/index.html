For ciklus I. 
    
    
      # A for ciklus egy folyamatvezérlő utasítás
      
      # Adott egy gyumi tömb
      
      gyumi = ['alma', 'körte', 'szilva', 'barack']
      
      #A for ciklus segítségével kiírjuk a gyumi tömb elemeit sorba
      # Az i a ciklus indexét is jelölheti azaz folyamatosan növeli magát 1-gyel!(Erről később bővebben lesz infó)
      # A range() jelöli, hogy meddig menjen a ciklus
      # A len(gyumi) megadja az adott tömb hosszát
      for i in range(len(gyumi)):
        print(gyumi[i])

For ciklus II. 
    
    
      # A for ciklus lépéseit meghatározó adatokból max. 3-at adhatunk meg:
      # 1 adat esetében a ciklus végét határozza meg
      # 2 adat esetében az első a ciklus elejét, a második a ciklus végét határozza meg
      # 3 adat esetében az első a ciklus elejét, a második a ciklus végét, a harmadik a lépés méretét határozza meg

      #1 adat, ciklus vége, 1-10 ig számol
      for i in range(10):
        print(i)
      
      #2 adat, a első a kezdés, a második a vége, 5-15 ig száol
      for i in range(5,15):
        print(i)

      #3 adat, a első a kezdés, a második a vége, 10-20 ig számol kettesével
      for i in range(10,20,2):
        print(i)
